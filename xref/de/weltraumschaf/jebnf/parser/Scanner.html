12  package de.weltraumschaf.jebnf.parser;
13  
14  import java.io.Closeable;
15  
16  /**
17   * Interface for scanner.
18   *
19   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
20   */
21  public interface Scanner extends CharacterAccess, TokenAccess, Closeable {
22  
23      /**
24       * Get the scanned file name.
25       *
26       * This may be any string, if not scanning from a real file.
27       *
28       * @return Return file name string.
29       */
30      String getFile();
31  
32      /**
33       * Creates a {Position} from the current line and column in the input stream.
34       *
35       * @return Return always new instance.
36       */
37      Position createPosition();
38  
39      /**
40       * Checks if the current character is a new line character (\n or \r)
41       * and if it is increments the line counter and resets the column counter to 0.
42       */
43      void checkNewline();
44  
45      /**
46       * Throws a {SyntaxException} with the current {Position} in the input stream.
47       *
48       * @param msg Describes raised error..
49       * @throws SyntaxException On syntax errors.
50       */
51      void raiseError(final String msg) throws SyntaxException;
52  
53  }
