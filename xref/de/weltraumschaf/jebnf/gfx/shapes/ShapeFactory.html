119     /**
120      * Creates one of the {@link Curves "curves"}.
121      *
122      * @param type Type of curve to create.
123      * @return Always return new instance.
124      */
125     public static Curve curve(final Curves type) {
126         Curve curve;
127 
128         switch (type) {
129             case NORTH_EAST:
130                 curve = new CurveNE();
131                 break;
132             case NORTH_WEST:
133                 curve = new CurveNW();
134                 break;
135             case SOUTH_EAST:
136                 curve = new CurveSE();
137                 break;
138             case SOUTH_WEST:
139                 curve = new CurveSW();
140                 break;
141             default:
142                 throw new IllegalArgumentException(String.format("Unsupported type: %s!", type));
143         }
144 
145         return curve;
146     }

165     /**
166      * Creates a fork railroad shape.
167      *
168      * A fork is a combination of a straight and curve.
169      *
170      * @param orientation Type of straight.
171      * @param curve Type of curve.
172      * @return Always return new instance.
173      */
174     public static Shape fork(final Straights orientation, final Curves curve) {
175         switch (orientation) {
176             case NORT_SOUTH:
177                 return verticalFork(curve);
178             case WEST_EAST:
179                 return horizontalFork(curve);
180             default:
181                 throw new IllegalArgumentException("Unsupported orientation: " + orientation + "!");
182         }
183     }
184
