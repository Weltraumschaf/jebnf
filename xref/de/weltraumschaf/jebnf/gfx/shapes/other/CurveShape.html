15  import de.weltraumschaf.jebnf.gfx.Size;
16  import de.weltraumschaf.jebnf.gfx.Strokes;
17  import java.awt.Color;
18  import java.awt.Graphics2D;
19  import java.awt.geom.Arc2D;

21  /**
22   * Draw curves.
23   *
24   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
25   */
26  public class CurveShape extends RectangularShape {
27  
28      /**
29       * Don't know why I need this. W/o it looks bad in output.
30       */
31      private static final int MAGIC = 1;
32  
33      /**
34       * Type of curve railroad shapes.
35       *
36       * @author Sven Strittmatter <weltraumschaf@googlemail.com>
37       */
38      public enum Directions {
39  
40          /**
41           * Curve railroad from north to west.
42           */
43          NORTH_WEST,
44          /**
45           * Curve railroad from north to east.
46           */
47          NORTH_EAST,
48          /**
49           * Curve railroad from south to west.
50           */
51          SOUTH_WEST,
52          /**
53           * Curve railroad from south to east.
54           */
55          SOUTH_EAST;
56  
57      }
58  
59      /**
60       * Start arc at 0 degree.
61       */
62      private static final int SOUTH = 0;
63  
64      /**
65       * Start arc at 90 degree.
66       */
67      private static final int EAST = 90;
68  
69      /**
70       * Start arc at 180 degree.
71       */
72      private static final int NORTH = 180;
73  
74      /**
75       * Start arc at 270 degree.
76       */
77      private static final int WEST = 270;
78  
79  
80      /**
81       * Arc extend to 90 degree.
82       */
83      private static final int QUARTER_CLOCKWISE = 90;
84  
85      /**
86       * Direction of curve arc (e.g. from south to east.
87       */
88      private final Directions direction;
89  
90      /**
91       * Initializes the curve with the direction type.
92       *
93       * @param direction Direction from where to start and end the curve.
94       */
95      public CurveShape(final Directions direction) {
96          this.direction = direction;
97      }
98  
99      @Override
100     public void paint(final Graphics2D graphic) {
101         super.paint(graphic);
102         backupColorAndStroke(graphic);
103         graphic.setStroke(Strokes.createForRail());
104         graphic.setColor(Color.BLACK);
105         graphic.draw(createArc());
106         resotreColorAndStroke(graphic);
107     }
108 
109     /**
110      * Creates the arc for the curve depending on {@link #direction}.
111      *
112      * @return Returns AWT arc object.
113      */
114     protected Arc2D createArc() {
115         final Point shapePosition = getPosition();
116         final Size size = getSize();
117         int startAngle;
118         Point arcPosition;
119         Size arcSize;
120 
121         switch (direction) {
122             case NORTH_EAST:
123                 startAngle = NORTH;
124                 arcPosition = Point.valueOf(shapePosition.getX() + size.getWidth() / 2,
125                                             shapePosition.getY() - (size.getHeight() / 2) - MAGIC);
126                 arcSize = getSize().setWidth(getSize().getWidth() + MAGIC);
127                 break;
128             case NORTH_WEST:
129                 startAngle = WEST;
130                 arcPosition = Point.valueOf(shapePosition.getX() - size.getWidth() / 2 - MAGIC,
131                                             shapePosition.getY() - (size.getHeight() / 2) - MAGIC);
132                 arcSize = getSize().setWidth(getSize().getWidth() - MAGIC);
133                 break;
134             case SOUTH_EAST:
135                 startAngle = EAST;
136                 arcPosition = Point.valueOf(shapePosition.getX() + size.getWidth() / 2,
137                                             shapePosition.getY() + size.getHeight() / 2);
138                 arcSize = Size.valueOf(getSize().getWidth() + MAGIC, getSize().getHeight() + MAGIC);
139                 break;
140             case SOUTH_WEST:
141                 startAngle = SOUTH;
142                 arcPosition = Point.valueOf(shapePosition.getX() - (size.getWidth() / 2) - MAGIC,
143                                             shapePosition.getY() + size.getHeight() / 2);
144                 arcSize = getSize().setHeight(getSize().getHeight() + MAGIC);
145                 break;
146             default:
147                 startAngle = -1;
148                 arcPosition = null;
149                 arcSize = null;
150                 break;
151         }
152 
153         if (-1 == startAngle) {
154             throw new IllegalArgumentException(String.format("Unsupported straight type: %s!", direction));
155         }
156 
157         return new Arc2D.Float(arcPosition.getX(),
158                                arcPosition.getY(),
159                                arcSize.getWidth(),
160                                arcSize.getHeight(),
161                                startAngle,
162                                QUARTER_CLOCKWISE,
163                                Arc2D.OPEN);
164     }
165 
166     /**
167      * Get the direction type.
168      *
169      * @return Enum type of direction.
170      */
171     public Directions getDirection() {
172         return direction;
173     }
