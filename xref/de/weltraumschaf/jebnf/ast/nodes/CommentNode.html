12  package de.weltraumschaf.jebnf.ast.nodes;
13  
14  import de.weltraumschaf.jebnf.ast.Node;
15  import de.weltraumschaf.jebnf.ast.NodeType;
16  import de.weltraumschaf.jebnf.ast.Notification;
17  
18  /**
19   * CommentNode node.
20   *
21   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
22   */
23  public final class CommentNode extends BaseNode {
24  
25      /**
26       * Key for the value attribute.
27       */
28      private static final String ATTR_VALUE = "value";
29  
30      /**
31       * Initializes object with empty value and parent node.
32       *
33       * @param parent The parent node.
34       * @param value CommentNode value.
35       */
36      private CommentNode(final Node parent, final String value) {
37          super(parent, NodeType.COMMENT);
38          setAttribute(ATTR_VALUE, value);
39      }
40  
41      /**
42       * Creates new comment node with {@link NullNode} parent node and empty string value.
43       *
44       * @return New instance.
45       */
46      public static CommentNode newInstance() {
47          return newInstance(NullNode.getInstance());
48      }
49  
50      /**
51       * Creates new comment node with {@link NullNode} parent node.
52       *
53       * @param value The comment string.
54       * @return       New instance.
55       */
56      public static CommentNode newInstance(final String value) {
57          return newInstance(NullNode.getInstance(), value);
58      }
59  
60      /**
61       * Creates new comment node with empty string value.
62       *
63       * @param parent The node's parent.
64       * @return        New instance.
65       */
66      public static CommentNode newInstance(final Node parent) {
67          return newInstance(parent, "");
68      }
69  
70      /**
71       * Creates new comment node.
72       *
73       * @param parent The node's parent.
74       * @param value  The comment string.
75       * @return        New instance.
76       */
77      public static CommentNode newInstance(final Node parent, final String value) {
78          return new CommentNode(parent, value);
79      }
80  
81      @Override
82      public void probeEquivalence(final Node other, final Notification result) {
83          try {
84              final CommentNode terminal = (CommentNode) other;
85  
86              if (!getAttribute(ATTR_VALUE).equals(terminal.getAttribute(ATTR_VALUE))) {
87                  result.error("Comment value mismatch: '%s' != '%s'!", getAttribute(ATTR_VALUE),
88                                                                        terminal.getAttribute(ATTR_VALUE));
89              }
90          } catch (ClassCastException ex) {
91              result.error(
92                  "Probed node types mismatch: '%s' != '%s'!",
93                  getClass(),
94                  other.getClass()
95              );
96          }
97      }
98  
99      /**
100      * Always returns 1 because a terminal has no child nodes..
101      *
102      * @return Returns 1.
103      */
104     @Override
105     public int depth() {
106         return 1;
107     }
108 
109 }
