32  /**
33   * Visits the given {@link de.weltraumschaf.jebnf.ast.nodes.SyntaxNode} and generates
34   * a {@link de.weltraumschaf.jebnf.gfx.RailroadDiagram} as result.
35   *
36   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
37   */
38  public class RailroadDiagramVisitor implements Visitor<RailroadDiagram> {
39  
40      /**
41       * Logging facility.
42       */
43      private static final Logger LOGGER = Logger.getLogger(RailroadDiagramVisitor.class.getName());
44  
45      /**
46       * Holds the resulting diagram.
47       */
48      private RailroadDiagram diagram;
49  
50      /**
51       * Stack of shapes currently created in {@link #visit(de.weltraumschaf.jebnf.ast.Visitable)}.
52       */
53      private final Stack<Shape> currentShapeStack = new Stack<Shape>();
54  
55      @Override
56      public void beforeVisit(final Visitable visitable) {
57          // Nothing to do here.
58      }
59  
60      @Override
61      public void visit(final Visitable visitable) {
62          final Node node      = (Node) visitable;
63          Shape shape          = empty();
64  
65          switch (node.getType()) {
66              case SYNTAX: {
67                  // Reset everything because it's the root node.
68                  currentShapeStack.removeAllElements();
69                  diagram = new RailroadDiagram();
70                  diagram.setTitle(node.getAttribute(SyntaxNode.ATTRIBUTE_TITLE));
71                  diagram.setMeta(node.getAttribute(SyntaxNode.ATTRIBUTE_META));
72                  break;
73              }
74              case RULE: {
75                  shape = rule(node.getAttribute(RuleNode.ATTRIBUTE_NAME));
76                  diagram.add(row(shape));
77                  shape = row(start());
78                  diagram.add(shape);
79                  break;
80              }
81              case CHOICE: {
82                  shape = choice();
83                  addShapeToCurrentShape(shape);
84                  break;
85              }
86              case LOOP: {
87                  shape = loop();
88                  addShapeToCurrentShape(shape);
89                  break;
90              }
91              case OPTION: {
92                  shape = option();
93                  addShapeToCurrentShape(shape);
94                  break;
95              }
96              case SEQUENCE: {
97                  shape = row();
98                  addShapeToCurrentShape(shape);
99                  break;
100             }
101             case IDENTIFIER: {
102                 shape = identifier(node.getAttribute(IdentifierNode.ATTRIBUTE_VALUE));
103                 addShapeToCurrentShape(shape);
104                 break;
105             }
106             case TERMINAL: {
107                 shape = terminal(node.getAttribute(TerminalNode.ATTR_IBUTE_VALUE));
108                 addShapeToCurrentShape(shape);
109                 break;
110             }
111             case COMMENT: {
112                 // For the moment we ignore comments.
113                 return;
114             }
115             case NULL: {
116                 // Ignore null nodes.
117                 return;
118             }
119             default:
120                 LOGGER.log(Level.WARNING, String.format("Unsupported node type: '%s'!", node.getType()));
121                 return;
122         }
123 
124         currentShapeStack.push(shape);
125     }
126 
127     @Override
128     public void afterVisit(final Visitable visitable) {
129         if (!currentShapeStack.isEmpty()) {
130             currentShapeStack.pop();
131         }
132     }
133 
134     @Override
135     public RailroadDiagram getResult() {
136         return diagram;
137     }
138 
139     /**
140      * Add a shape to the shape on top of the {@link #currentShapeStack}.
141      *
142      * @param shape Shape to add.
143      */
144     private void addShapeToCurrentShape(final Shape shape) {
145         final Shape currentShape = currentShapeStack.peek();
146 
147         if (currentShape instanceof RowLayoutShape) {
148             ((RowLayoutShape) currentShape).append(shape);
149         } else if (currentShape instanceof ChoiceShape) {
150             ((ChoiceShape) currentShape).addChoice(shape);
151         } else if (currentShape instanceof LoopShape) {
152             ((LoopShape) currentShape).setLooped(shape);
153         } else if (currentShape instanceof OptionShape) {
154             ((OptionShape) currentShape).setOptional(shape);
155         } else if (currentShape instanceof SequenceShape) {
156             ((SequenceShape) currentShape).append(shape);
157         }
158     }
