12  package de.weltraumschaf.jebnf.gfx.shapes.compound;
13  
14  import com.google.common.collect.Lists;
15  import de.weltraumschaf.jebnf.gfx.Point;
16  import de.weltraumschaf.jebnf.gfx.Size;
17  import de.weltraumschaf.jebnf.gfx.shapes.Adjustable;
18  import de.weltraumschaf.jebnf.gfx.shapes.Shape;
19  import java.awt.Graphics2D;
20  import java.util.List;

22  /**
23   * A grid layout organizes multiple {@link ColumnLayout "column layouts"} to a two dimensional grid.
24   *
25   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
26   */
27  public class GridLayout extends AbstractLayout implements Adjustable {
28  
29      /**
30       * List of column layouts.
31       */
32      private final List<ColumnLayout> columns = Lists.newArrayList();
33  
34      /**
35       * Count the containing columns of the grid.
36       *
37       * @return Return the number of columns.
38       */
39      public int countCols() {
40          return columns.size();
41      }
42  
43      /**
44       * Count the rows of the grid.
45       *
46       * @return Return the number of rows.
47       */
48      public int counRows() {
49          return countCols() == 0
50                 ? 0
51                 : columns.get(0).countShapes();
52      }
53  
54      /**
55       * Get the column at given index.
56       *
57       * @param columnIndex Index beginning with 0.
58       * @return Return the column.
59       */
60      public ColumnLayout get(final int columnIndex) {
61          ColumnLayout col;
62  
63          try {
64              col = columns.get(columnIndex);
65          } catch (IndexOutOfBoundsException ex) {
66              col = null;
67          }
68  
69          if (null == col) {
70              throw new IllegalArgumentException(
71                      String.format("The column at columnIndex %d is not present!", columnIndex));
72          }
73  
74          return col;
75      }
76  
77      /**
78       * Get shape at given column and row index.
79       *
80       * @param columnIndex Index beginning with 0.
81       * @param rowIndex Index beginning with 0.
82       * @return Return the shape.
83       */
84      public Shape get(final int columnIndex, final int rowIndex) {
85          Shape shape;
86  
87          try {
88              shape = get(columnIndex).get(rowIndex);
89          } catch (IndexOutOfBoundsException ex) {
90              shape = null;
91          }
92  
93          if (null == shape) {
94              throw new IllegalArgumentException(
95                      String.format("The shape at columnIndex %d and rowIndex %d is not present!",
96                                    columnIndex, rowIndex));
97          }
98  
99          return shape;
100     }
101 
102     /**
103      * Set shape at given column and row index.
104      *
105      * @param columnIndex Index beginning with 0.
106      * @param rowIndex Index beginning with 0.
107      * @param shape Shape to set.
108      * @return Return itself for method chaining.
109      */
110     public GridLayout set(final int columnIndex, final int rowIndex, final Shape shape) {
111         final int colCount = countCols();
112 
113         if (columnIndex >= colCount) {
114             for (int i = colCount; i < columnIndex + 1; ++i) {
115                 columns.add(ColumnLayout.newColumnLayout());
116             }
117         }
118 
119         final ColumnLayout col = columns.get(columnIndex);
120         col.set(rowIndex, shape);
121         return this;
122     }
123 
124     /**
125      * Append columns to the grid.
126      *
127      * @param cols Columns to add.
128      * @return Return itself for method chaining.
129      */
130     public GridLayout append(final ColumnLayout... cols) {
131         for (ColumnLayout cl : cols) {
132             append(cl);
133         }
134 
135         return this;
136     }
137 
138     /**
139      * Append column to the grid.
140      *
141      * @param col Column to add.
142      * @return Return itself for method chaining.
143      */
144     public GridLayout append(final ColumnLayout col) {
145         columns.add(col);
146         return this;
147     }
148 
149     @Override
150     public void paint(final Graphics2D graphic) {
151         final Point pos = getPosition();
152         int currentX = pos.getX();
153 
154         for (ColumnLayout col : columns) {
155             col.setPosition(pos.setX(currentX));
156             col.setDebug(isDebug());
157             col.paint(graphic);
158             currentX += col.getSize().getWidth();
159         }
160     }
161 
162     @Override
163     public void adjust(final Graphics2D graphic) {
164         int width = 0;
165         int height = 0;
166 
167         for (ColumnLayout col : columns) {
168             col.adjust(graphic);
169             final  Size colSize = col.getSize();
170             width  += colSize.getWidth();
171             height = Math.max(height, colSize.getHeight());
172         }
173 
174         setSize(new Size(width, height));
175     }
